33void AirQualityManager::fetchStations() {
34 QUrl url("https://api.gios.gov.pl/pjp-api/rest/station/findAll");
35 QNetworkReply *reply = networkManager->get(QNetworkRequest(url));
36 connect(reply, &QNetworkReply::finished, this, [this, reply]() {
37 if (reply->error() == QNetworkReply::NoError) {
38 QList<Station> stations;
39 QJsonDocument doc = QJsonDocument::fromJson(reply->readAll());
40 QJsonArray array = doc.array();
41 for (const QJsonValue &value : array) {
42 QJsonObject obj = value.toObject();
43 Station s;
44 s.id = obj["id"].toInt();
45 s.name = obj["stationName"].toString();
46 s.latitude = obj["gegrLat"].toString().toDouble();
47 s.longitude = obj["gegrLon"].toString().toDouble();
48 stations.append(s);
49 }
50 emit stationsFetched(stations);
51 } else {
52 emit errorOccurred(reply->errorString());
53 }
54 reply->deleteLater();

94void AirQualityManager::fetchSensorData(int sensorId) {
95 QUrl url(QString("https://api.gios.gov.pl/pjp-api/rest/data/getData/%1").arg(sensorId));
96 QNetworkReply *reply = networkManager->get(QNetworkRequest(url));
97 connect(reply, &QNetworkReply::finished, this, [this, reply, sensorId]() {
98 if (reply->error() == QNetworkReply::NoError) {
99 QList<Measurement> measurements;
100 QJsonDocument doc = QJsonDocument::fromJson(reply->readAll());
101 QJsonObject obj = doc.object();
102 QJsonArray values = obj["values"].toArray();
103 QString paramName = obj["key"].toString();
104
105 for (const QJsonValue &value : values) {
106 QJsonObject measurementObj = value.toObject();
107 Measurement m;
108 m.stationId = -1;
109 m.paramName = paramName;
110 m.value = measurementObj["value"].toDouble(-1.0);
111 m.dateTime = QDateTime::fromString(measurementObj["date"].toString(), Qt::ISODate);
112 measurements.append(m);
113 }
114
115 if (!measurements.isEmpty() && measurements.first().stationId != -1) {
116 int stationId = measurements.first().stationId;
117 onlineMeasurements[stationId] = measurements;
118 }
119
120 emit measurementsFetched(measurements);
121 } else {
122 emit errorOccurred(reply->errorString());
123 }
124 reply->deleteLater();
125 });
