112QPointF MapWidget::latLonToPixel(double latitude, double longitude) {
113 if (!mapLoaded) {
114 return QPointF(0, 0);
115 }
116
117 const double minLat = 49.0, maxLat = 54.8;
118 const double minLon = 14.1, maxLon = 24.1;
119
120 double xRatio = (longitude - minLon) / (maxLon - minLon);
121 double yRatio = 1.0 - (latitude - minLat) / (maxLat - minLat);
122
123 double x = xRatio * mapImage.width();
