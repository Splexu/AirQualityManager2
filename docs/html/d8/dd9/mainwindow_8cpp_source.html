414double MainWindow::calculateDistance(double lat1, double lon1, double lat2, double lon2) {
415 const double PI = 3.141592653589;
416 const double R = 6371.0;
417
418 lat1 = lat1 * PI / 180.0;
419 lon1 = lon1 * PI / 180.0;
420 lat2 = lat2 * PI / 180.0;
421 lon2 = lon2 * PI / 180.0;
422
423 double dLat = lat2 - lat1;
424 double dLon = lon2 - lon1;
425
426 double a = sin(dLat / 2.0) * sin(dLat / 2.0) +
427 cos(lat1) * cos(lat2) * sin(dLon / 2.0) * sin(dLon / 2.0);
428 double c = 2.0 * atan2(sqrt(a), sqrt(1.0 - a));
429 double distance = R * c;
430
431 return distance;
432}

438void MainWindow::analyzeMeasurements(const QList<Measurement> &measurements) {
439 if (measurements.isEmpty()) {
440 analysisTextEdit->setText("Brak danych do analizy.");
441 return;
442 }
443
444 double minValue = measurements.first().value;
445 double maxValue = measurements.first().value;
446 QDateTime minDateTime = measurements.first().dateTime;
447 QDateTime maxDateTime = measurements.first().dateTime;
448 double sum = 0.0;
449 int validCount = 0;
450
451 for (const auto &m : measurements) {
452 if (m.value >= 0) {
453 if (m.value < minValue) {
454 minValue = m.value;
455 minDateTime = m.dateTime;
456 }
457 if (m.value > maxValue) {
458 maxValue = m.value;
459 maxDateTime = m.dateTime;
460 }
461 sum += m.value;
462 validCount++;
463 }
464 }
465
466 double average = (validCount > 0) ? sum / validCount : 0.0;
467
468 double trend = 0.0;
469 if (validCount > 1) {
470 double sumX = 0.0, sumY = 0.0, sumpXY = 0.0, sumXX = 0.0;
471 int i = 0;
472 for (const auto &m : measurements) {
473 if (m.value >= 0) {
474 double x = i;
475 double y = m.value;
476 sumX += x;
477 sumY += y;
478 sumXY += x * y;
479 sumXX += x * x;
480 i++;
481 }
482 }
483 double n = validCount;
484 trend = (n * sumXY - sumX * sumY) / (n * sumXX - sumX * sumX);
485 }
486
487 QString analysisText = "Analiza danych:\n";
488 analysisText += QString("Najmniejsza wartość: %1 (%2)\n").arg(minValue).arg(minDateTime.toString());
489 analysisText += QString("Największa wartość: %1 (%2)\n").arg(maxValue).arg(maxDateTime.toString());
490 analysisText += QString("Średnia wartość: %1\n").arg(average, 0, 'f', 2);
491 analysisText += "Trend: ";
492 if (trend > 0) {
493 analysisText += "Rosnący";
494 } else if (trend < 0) {
495 analysisText += "Malejący";
496 } else {
497 analysisText += "Stabilny";
498 }
499
500 analysisTextEdit->setText(analysisText);
501}

654void MainWindow::updateChart(const QList<Measurement> &measurements) {
655 series->clear();
656
657 int days = periodComboBox->currentData().toInt();
658 QDateTime now = QDateTime::currentDateTime();
659 QDateTime cutoff = now.addDays(-days);
660
661 double minValue = std::numeric_limits<double>::max();
662 double maxValue = std::numeric_limits<double>::lowest();
663
664 for (const auto &m : measurements) {
665 if (m.value >= 0 && m.dateTime >= cutoff) {
666 series->append(m.dateTime.toMSecsSinceEpoch(), m.value);
667 minValue = qMin(minValue, m.value);
668 maxValue = qMax(maxValue, m.value);
669 }
670 }
671
672 if (series->count() > 0) {
673 QChart *chart = chartView->chart();
674 QDateTimeAxis *axisX = qobject_cast<QDateTimeAxis*>(chart->axisX());
675 QValueAxis *axisY = qobject_cast<QValueAxis*>(chart->axisY());
676
677 axisX->setRange(cutoff, now);
678 axisY->setRange(minValue * 0.9, maxValue * 1.1);
679 chart->setTitle("Wykres pomiarów: " + measurements.first().paramName);
680 } else {
681 chartView->chart()->setTitle("Brak danych dla wybranego okresu");
682 }
